CREATE OR REPLACE TRIGGER RELLENARTABLACONTROLJOCKEYS

DECLARE
  CURSOR C_DNIFECHA IS
    SELECT P.DNIJOCKEY, C.FECHAHORA

    WHERE P.CODIGOCARRERA = C.CODIGOCARRERA;
